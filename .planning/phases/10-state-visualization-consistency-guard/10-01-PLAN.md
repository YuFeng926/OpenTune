     * RenderStatus - Snapshot-driven render status (STAB-02)

     * - Idle: No chunks, no work pending (empty or all blank)
     * - Dirty: Has chunks, but desiredRevision != publishedRevision for some
     * - Rendering: pending > 0 || running > 0
     * - Ready: All chunks idle with matching revisions, total > 0

    enum class RenderStatus {

        Rendering,  // Active render in progress
        Ready,      // All content rendered and ready
        Error       // Render failure (future)
    };

        RenderCache::ChunkStats chunkStats;

    ```cpp
    RenderStatusSnapshot OpenTuneAudioProcessorEditor::getRenderStatusSnapshot() const
    {
        RenderStatusSnapshot snapshot;

        uint64_t clipId = resolveCurrentClipId();
        snapshot.clipId = clipId;

        if (clipId == 0) {
            snapshot.status = RenderStatus::Idle;
            return snapshot;
        }

        auto chunkStats = processorRef_.getClipChunkStatsById(kVst3TrackId, clipId);
        snapshot.chunkStats = chunkStats;
        snapshot.hasContent = chunkStats.total() > 0;

        // Determine status from snapshot data
        if (chunkStats.hasActiveWork()) {
            snapshot.status = RenderStatus::Rendering;
        } else if (chunkStats.total() == 0) {
            snapshot.status = RenderStatus::Idle;
        } else if (chunkStats.idle > 0 && chunkStats.pending == 0 && chunkStats.running == 0) {
            snapshot.status = RenderStatus::Ready;
        } else {
            snapshot.status = RenderStatus::Dirty;
        }

        return snapshot;
    }
    ```
