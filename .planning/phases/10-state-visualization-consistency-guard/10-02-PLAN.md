    ```cpp
    // Diagnostic Info API (STAB-03)

    /**
     * DiagnosticInfo - Minimal diagnostic snapshot for troubleshooting
     *
     * Provides observable fields for consistency verification:
     * - editVersion: Current edit version counter
     * - publishedRevision: Latest published revision for clip
     * - lastControlCall: Last transport control requested
     * - lastControlTimestamp: When last control was requested
     */
    struct DiagnosticInfo {
        int editVersion{0};
        uint64_t publishedRevision{0};
        uint64_t desiredRevision{0};
        juce::String lastControlCall;
        juce::int64 lastControlTimestamp{0};  // milliseconds since epoch
        int totalChunks{0};
        int pendingChunks{0};
        int runningChunks{0};
    };

    /**
     * Get diagnostic info for a track
     * @param trackId Track ID (default 0 for VST3 single-track)
     * @return Diagnostic snapshot
     */
    DiagnosticInfo getDiagnosticInfo(int trackId = 0) const;
    ```

    ```cpp
    OpenTuneAudioProcessor::DiagnosticInfo OpenTuneAudioProcessor::getDiagnosticInfo(int trackId) const
    {
        DiagnosticInfo info;

        // Edit version from parameter
        if (editVersionParam_ != nullptr) {
            info.editVersion = editVersionParam_->get();
        }

        // Last control call info
        const int controlType = lastControlType_.load(std::memory_order_relaxed);
        switch (controlType) {
            case 1: info.lastControlCall = "play"; break;
            case 2: info.lastControlCall = "pause"; break;
            case 3: info.lastControlCall = "stop"; break;
            case 4: info.lastControlCall = "seek"; break;
            default: info.lastControlCall = "none"; break;
        }
        info.lastControlTimestamp = lastControlTimestamp_.load(std::memory_order_relaxed);

        // Chunk stats from render cache
        const juce::ScopedReadLock rl(tracksLock_);
        if (trackId >= 0 && trackId < MAX_TRACKS && !tracks_[trackId].clips.empty()) {
            const auto& clip = tracks_[trackId].clips[tracks_[trackId].selectedClipIndex >= 0 
                                                       ? tracks_[trackId].selectedClipIndex 
                                                       : 0];
            if (clip.renderCache) {
                auto stats = clip.renderCache->getChunkStats();
                info.totalChunks = stats.total();
                info.pendingChunks = stats.pending;
                info.runningChunks = stats.running;

                // Get latest published/desired revision from chunks
                // (simplified: use max revision across chunks)
                auto published = clip.renderCache->getPublishedChunks();
                if (!published.empty()) {
                    // Use first chunk's revision as representative
                    // Note: In production, might want max or most recent
                }
                // For now, desiredRevision can be obtained from a pending job if available
                RenderCache::PendingJob job;
                if (clip.renderCache->getNextPendingJob(job)) {
                    info.desiredRevision = job.targetRevision;
                }
            }
        }

        return info;
    }
    ```
